 // Update User
        public async Task<IActionResult> UpdateUser(int id, User user)
        {
            if (id != user.UserId)
            {
                return BadRequest();
            }

            _context.Entry(user).State = EntityState.Modified;

            try
            {
               await _userService.UpdateAsync(user);
            }
            catch (NotFoundException)
            {

                return NotFound();

            }
            catch(DbUpdateConcurrencyException)
            {
                return BadRequest();
            }

            return NoContent();

        }
